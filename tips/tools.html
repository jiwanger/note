进制转换工具

程序下载： number-format

功能描述：本程序接收一个 二进制/十制整/十六进制 正整数值参数，并显示其 十六进制 & 十制整 & 八制整 & 二进制 形式。

编译运行：本程序为命令行工具，用 C 语言实现，在 Linux Mint 32 位中编译运行。 

$ gcc src/main/number-format.c -o jnumfmt -std=c11

# 参数：
#	参数可以使用空格, 如: 0x0F FE, 0b11 1000...
#	参数为二进制正整数值，要加前缀 0b, 如: 0b100, 0b111, 0b11 1000...
#	参数为十进制正整数值，直接输入, 如: 32, 1000, 429 496 729...
#	参数为十六进制正整数值，要加前缀 0x, 如: 0x32, 0xFF, 0xFF FF FF FF...
$ ./jnumfmt 0b10			# 二进制测试
$ ./jnumfmt 0b1001 1001		# 二进制测试
$ ./jnumfmt 1024			# 十进制测试
$ ./jnumfmt 10 240			# 十进制测试
$ ./jnumfmt 0xFF			# 十六进制测试
$ ./jnumfmt 0xA0 FF			# 十六进制测试

#include <ctype.h>
#include <stdio.h>
#include <stdlib.h>
#include <string.h>

#define ARG_ERROR_MESSAGE "非法参数，必须接收一个 二进制/十制整/十六进制 正整数值参数,\
    如： 32, 1000, 0b100, 0b111, 0x32, 0xA0, 0xF003...\n"

void delete_blank(char *);
/**
 *  本程序接收一个 二进制/十制整/十六进制 正整数值参数，并显示其 十六进制 & 十制整 & 八制整 & 二进制 形式。
 *  长参数可以使用空格, 如: 0x0F FE, 0b11 1000...
 *  参数为二进制正整数值，要加前缀 0b, 如: 0b100, 0b111, 0b111000...
 *  参数为十进制正整数值，直接输入, 如: 32, 1000, 429496729...
 *  参数为十六进制正整数值，要加前缀 0x, 如: 0x32, 0xFF, 0xFFFFFFFF...
 */
int main(int argc, char* argv[])
{
    if (argc < 2 || !strcmp("-h", argv[1]) || !strcmp("--help", argv[1])) {
        printf("本函数必须接收一个最大 0xFFFFFFFF 的 二进制/十制整/十六进制 正整数值参数，并显示其各种形式！\n");
        printf("  -- 长参数可以使用空格, 如: 0x0F FE, 0b11 1000...\n");
        printf("  -- 参数为二进制正整数值，要加前缀 0b, 如: 0b100, 0b111, 0b111000...\n");
        printf("  -- 参数为十进制正整数值，直接输入, 如: 32, 1000, 429496729...\n");
        printf("  -- 参数为十六进制正整数值，要加前缀 0x, 如: 0x32, 0xFF, 0xFFFFFFFF...\n");
        return 0;
    }

    char input[50];// 参数最长为: 0b0000 0000 1000 0000 0000 0000 0000 0000
    memset(input, '\0', sizeof(input));
    for(int k = 1; k < argc; k++) {
        strcat(strcat(input, " "), argv[k]);
    }
    
    char arg[50];
    strcpy(arg, input);
    delete_blank(arg);

    int len = strlen(arg);
 
    int flag = 10; // 输入参数为十制整/十六进制
 
    if (len > 2 && !strncmp("0x", arg, 2)) { // 十六进制整
        for (int k = 2; k < len; k++) {
            if (!isxdigit(arg[k])) {
                printf(ARG_ERROR_MESSAGE);
                return 0;
            }
        }
        flag = 16;
    } else if(len > 2 && !strncmp("0b", arg, 2)) {
        for (int k = 2; k < len; k++) {
            if ('1' != arg[k] && '0' != arg[k]) {
                printf(ARG_ERROR_MESSAGE);
                return 0;
            }
        }
        memset(arg, ' ', 2);
        flag = 2;
    } else { // 十制整
        for (int k = 0; k < len; k++) {
            if (!isdigit(arg[k])) {
                printf(ARG_ERROR_MESSAGE);
                return 0;
            }
        }
    }
 
    printf("输入的参数为 %d 进制： %s\n", flag, input);
    unsigned long n = strtoul(arg, NULL, flag);
 
    printf("十六进制:  %lX\n", n);
    printf("  十进制:  %lu\n", n);
    printf("  八进制:  %lo\n", n);
 
    int i = 0;
    char bin[32];
 
    memset(bin, '0', 32);
 
    while (n) {
        bin[31 - i] = (n & 1) + '0';
        n = n >> 1;
        i++;
    }
    printf("  二进制: ");
    for (i = 0; i < 32; i++) {
        if (i % 4 == 0) {
            printf(" ");
        }
        printf("%c", bin[i]);
    }
    printf("\n");
 
    return 0;
}

/** 删除字符串的所有空格 */
void delete_blank(char s[])
{
    char *p = s;
    while (*p != '\0') {
        if (*p != ' ') {
            *s++ = *p;
        }
        p++;
    }
    *s = '\0';
}

001package org.junit.runners.model;
002
003import java.lang.reflect.Modifier;
004import java.util.List;
005
006/**
007 * Parent class for {@link FrameworkField} and {@link FrameworkMethod}
008 *
009 * @since 4.7
010 */
011public abstract class FrameworkMember<T extends FrameworkMember<T>> implements
012        Annotatable {
013    abstract boolean isShadowedBy(T otherMember);
014
015    /**
016     * Check if this member is shadowed by any of the given members. If it
017     * is, the other member is removed.
018     * 
019     * @return member that should be used, or {@code null} if no member should be used.
020     */
021    abstract T handlePossibleBridgeMethod(List<T> members);
022
023    abstract boolean isBridgeMethod();
024
025    protected abstract int getModifiers();
026
027    /**
028     * Returns true if this member is static, false if not.
029     */
030    public boolean isStatic() {
031        return Modifier.isStatic(getModifiers());
032    }
033
034    /**
035     * Returns true if this member is public, false if not.
036     */
037    public boolean isPublic() {
038        return Modifier.isPublic(getModifiers());
039    }
040
041    public abstract String getName();
042
043    public abstract Class<?> getType();
044
045    public abstract Class<?> getDeclaringClass();
046}

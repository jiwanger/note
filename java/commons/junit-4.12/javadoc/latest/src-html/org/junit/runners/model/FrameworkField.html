001package org.junit.runners.model;
002
003import java.lang.annotation.Annotation;
004import java.lang.reflect.Field;
005import java.util.List;
006
007import org.junit.runners.BlockJUnit4ClassRunner;
008
009/**
010 * Represents a field on a test class (currently used only for Rules in
011 * {@link BlockJUnit4ClassRunner}, but custom runners can make other uses)
012 *
013 * @since 4.7
014 */
015public class FrameworkField extends FrameworkMember<FrameworkField> {
016    private final Field field;
017
018    FrameworkField(Field field) {
019        if (field == null) {
020            throw new NullPointerException(
021                    "FrameworkField cannot be created without an underlying field.");
022        }
023        this.field = field;
024    }
025
026    @Override
027    public String getName() {
028        return getField().getName();
029    }
030
031    public Annotation[] getAnnotations() {
032        return field.getAnnotations();
033    }
034
035    public <T extends Annotation> T getAnnotation(Class<T> annotationType) {
036        return field.getAnnotation(annotationType);
037    }
038
039    @Override
040    public boolean isShadowedBy(FrameworkField otherMember) {
041        return false;
042    }
043
044    @Override
045    FrameworkField handlePossibleBridgeMethod(List<FrameworkField> members) {
046        return this; // fields are never bridge methods
047    }
048
049    @Override
050    boolean isBridgeMethod() {
051        return false;
052    }
053
054    @Override
055    protected int getModifiers() {
056        return field.getModifiers();
057    }
058
059    /**
060     * @return the underlying java Field
061     */
062    public Field getField() {
063        return field;
064    }
065
066    /**
067     * @return the underlying Java Field type
068     * @see java.lang.reflect.Field#getType()
069     */
070    @Override
071    public Class<?> getType() {
072        return field.getType();
073    }
074    
075    @Override
076    public Class<?> getDeclaringClass() {
077        return field.getDeclaringClass();
078    }
079
080    /**
081     * Attempts to retrieve the value of this field on {@code target}
082     */
083    public Object get(Object target) throws IllegalArgumentException, IllegalAccessException {
084        return field.get(target);
085    }
086
087    @Override
088    public String toString() {
089        return field.toString();
090    }
091}

048/**
049 * Provides most of the functionality specific to a Runner that implements a
050 * "parent node" in the test tree, with children defined by objects of some data
051 * type {@code T}. (For {@link BlockJUnit4ClassRunner}, {@code T} is
052 * {@link Method} . For {@link Suite}, {@code T} is {@link Class}.) Subclasses
053 * must implement finding the children of the node, describing each child, and
054 * running each child. ParentRunner will filter and sort children, handle
055 * {@code @BeforeClass} and {@code @AfterClass} methods,
056 * handle annotated {@link ClassRule}s, create a composite
057 * {@link Description}, and run children sequentially.
058 *
059 * @since 4.5
060 */
061public abstract class ParentRunner<T> extends Runner implements Filterable,
062        Sortable {
063    private static final List<TestClassValidator> VALIDATORS = Arrays.<TestClassValidator>asList(
064            new AnnotationsValidator());
065
066    private final Lock childrenLock = new ReentrantLock();
067    private final TestClass testClass;
068
069    // Guarded by childrenLock
070    private volatile Collection<T> filteredChildren = null;
071
072    private volatile RunnerScheduler scheduler = new RunnerScheduler() {
073        public void schedule(Runnable childStatement) {
074            childStatement.run();
075        }
076
077        public void finished() {
078            // do nothing
079        }
080    };

082    /**
083     * Constructs a new {@code ParentRunner} that will run {@code @TestClass}
084     */
085    protected ParentRunner(Class<?> testClass) throws InitializationError {
086        this.testClass = createTestClass(testClass);
087        validate();
088    }

096       this.testClass = notNull(testClass);

124    /**
125     * Runs the test corresponding to {@code child}, which can be assumed to be

136    /**
137     * Adds to {@code errors} a throwable for each problem noted with the test class (available from {@link #getTestClass()}).
138     * Default implementation adds an error for each method annotated with
139     * {@code @BeforeClass} or {@code @AfterClass} that is not
140     * {@code public static void} with no arguments.
141     */
142    protected void collectInitializationErrors(List<Throwable> errors) {
143        validatePublicVoidNoArgMethods(BeforeClass.class, true, errors);
144        validatePublicVoidNoArgMethods(AfterClass.class, true, errors);
145        validateClassRules(errors);
146        applyValidators(errors);
147    }
148
149    private void applyValidators(List<Throwable> errors) {
150        if (getTestClass().getJavaClass() != null) {
151            for (TestClassValidator each : VALIDATORS) {
152                errors.addAll(each.validateTestClass(getTestClass()));
153            }
154        }
155    }
156
157    /**
158     * Adds to {@code errors} if any method in this class is annotated with
159     * {@code annotation}, but:
160     * <ul>
161     * <li>is not public, or
162     * <li>takes parameters, or
163     * <li>returns something other than void, or
164     * <li>is static (given {@code isStatic is false}), or
165     * <li>is not static (given {@code isStatic is true}).
166     * </ul>
167     */
168    protected void validatePublicVoidNoArgMethods(Class<? extends Annotation> annotation,
169            boolean isStatic, List<Throwable> errors) {
170        List<FrameworkMethod> methods = getTestClass().getAnnotatedMethods(annotation);
171
172        for (FrameworkMethod eachTestMethod : methods) {
173            eachTestMethod.validatePublicVoidNoArg(isStatic, errors);
174        }
175    }
176
177    private void validateClassRules(List<Throwable> errors) {
178        CLASS_RULE_VALIDATOR.validate(getTestClass(), errors);
179        CLASS_RULE_METHOD_VALIDATOR.validate(getTestClass(), errors);
180    }

182    /**
183     * Constructs a {@code Statement} to run all of the tests in the test class.
184     * Override to add pre-/post-processing. Here is an outline of the
185     * implementation:
186     * <ol>
187     * <li>Determine the children to be run using {@link #getChildren()}
188     * (subject to any imposed filter and sort).</li>
189     * <li>If there are any children remaining after filtering and ignoring,
190     * construct a statement that will:
191     * <ol>
192     * <li>Apply all {@code ClassRule}s on the test-class and superclasses.</li>
193     * <li>Run all non-overridden {@code @BeforeClass} methods on the test-class
194     * and superclasses; if any throws an Exception, stop execution and pass the
195     * exception on.</li>
196     * <li>Run all remaining tests on the test-class.</li>
197     * <li>Run all non-overridden {@code @AfterClass} methods on the test-class
198     * and superclasses: exceptions thrown by previous steps are combined, if
199     * necessary, with exceptions from AfterClass methods into a
200     * {@link org.junit.runners.model.MultipleFailureException}.</li>
201     * </ol>
202     * </li>
203     * </ol>
204     *
205     * @return {@code Statement}
206     */
207    protected Statement classBlock(final RunNotifier notifier) {
208        Statement statement = childrenInvoker(notifier);
209        if (!areAllChildrenIgnored()) {
210            statement = withBeforeClasses(statement);
211            statement = withAfterClasses(statement);
212            statement = withClassRules(statement);
213        }
214        return statement;
215    }
216
217    private boolean areAllChildrenIgnored() {
218        for (T child : getFilteredChildren()) {
219            if (!isIgnored(child)) {
220                return false;
221            }
222        }
223        return true;
224    }
225
226    /**
227     * Returns a {@link Statement}: run all non-overridden {@code @BeforeClass} methods on this class
228     * and superclasses before executing {@code statement}; if any throws an
229     * Exception, stop execution and pass the exception on.
230     */
231    protected Statement withBeforeClasses(Statement statement) {
232        List<FrameworkMethod> befores = testClass
233                .getAnnotatedMethods(BeforeClass.class);
234        return befores.isEmpty() ? statement :
235                new RunBefores(statement, befores, null);
236    }
237
238    /**
239     * Returns a {@link Statement}: run all non-overridden {@code @AfterClass} methods on this class
240     * and superclasses after executing {@code statement}; all AfterClass methods are
241     * always executed: exceptions thrown by previous steps are combined, if
242     * necessary, with exceptions from AfterClass methods into a
243     * {@link org.junit.runners.model.MultipleFailureException}.
244     */
245    protected Statement withAfterClasses(Statement statement) {
246        List<FrameworkMethod> afters = testClass
247                .getAnnotatedMethods(AfterClass.class);
248        return afters.isEmpty() ? statement :
249                new RunAfters(statement, afters, null);
250    }
251
252    /**
253     * Returns a {@link Statement}: apply all
254     * static fields assignable to {@link TestRule}
255     * annotated with {@link ClassRule}.
256     *
257     * @param statement the base statement
258     * @return a RunRules statement if any class-level {@link Rule}s are
259     *         found, or the base statement
260     */
261    private Statement withClassRules(Statement statement) {
262        List<TestRule> classRules = classRules();
263        return classRules.isEmpty() ? statement :
264                new RunRules(statement, classRules, getDescription());
265    }
266
267    /**
268     * @return the {@code ClassRule}s that can transform the block that runs
269     *         each method in the tested class.
270     */
271    protected List<TestRule> classRules() {
272        List<TestRule> result = testClass.getAnnotatedMethodValues(null, ClassRule.class, TestRule.class);
273        result.addAll(testClass.getAnnotatedFieldValues(null, ClassRule.class, TestRule.class));
274        return result;
275    }
276
277    /**
278     * Returns a {@link Statement}: Call {@link #runChild(Object, RunNotifier)}
279     * on each object returned by {@link #getChildren()} (subject to any imposed
280     * filter and sort)
281     */
282    protected Statement childrenInvoker(final RunNotifier notifier) {
283        return new Statement() {
284            @Override
285            public void evaluate() {
286                runChildren(notifier);
287            }
288        };
289    }
290
291    /**
292     * Evaluates whether a child is ignored. The default implementation always
293     * returns <code>false</code>.
294     * 
295     * <p>{@link BlockJUnit4ClassRunner}, for example, overrides this method to
296     * filter tests based on the {@link Ignore} annotation.
297     */
298    protected boolean isIgnored(T child) {
299        return false;
300    }
301
302    private void runChildren(final RunNotifier notifier) {
303        final RunnerScheduler currentScheduler = scheduler;
304        try {
305            for (final T each : getFilteredChildren()) {
306                currentScheduler.schedule(new Runnable() {
307                    public void run() {
308                        ParentRunner.this.runChild(each, notifier);
309                    }
310                });
311            }
312        } finally {
313            currentScheduler.finished();
314        }
315    }

335    /**
336     * Runs a {@link Statement} that represents a leaf (aka atomic) test.
337     */
338    protected final void runLeaf(Statement statement, Description description,
339            RunNotifier notifier) {
340        EachTestNotifier eachNotifier = new EachTestNotifier(notifier, description);
341        eachNotifier.fireTestStarted();
342        try {
343            statement.evaluate();
344        } catch (AssumptionViolatedException e) {
345            eachNotifier.addFailedAssumption(e);
346        } catch (Throwable e) {
347            eachNotifier.addFailure(e);
348        } finally {
349            eachNotifier.fireTestFinished();
350        }
351    }

365    @Override
366    public Description getDescription() {
367        Class<?> clazz = getTestClass().getJavaClass();
368        Description description;
369        // if subclass overrides `getName()` then we should use it
370        // to maintain backwards compatibility with JUnit 4.12
371        if (clazz == null || !clazz.getName().equals(getName())) {
372            description = Description.createSuiteDescription(getName(), getRunnerAnnotations());
373        } else {
374            description = Description.createSuiteDescription(clazz, getRunnerAnnotations());
375        }
376
377        for (T child : getFilteredChildren()) {
378            description.addChild(describeChild(child));
379        }
380        return description;
381    }
382
383    @Override
384    public void run(final RunNotifier notifier) {
385        EachTestNotifier testNotifier = new EachTestNotifier(notifier,
386                getDescription());
387        testNotifier.fireTestSuiteStarted();
388        try {
389            Statement statement = classBlock(notifier);
390            statement.evaluate();
391        } catch (AssumptionViolatedException e) {
392            testNotifier.addFailedAssumption(e);
393        } catch (StoppedByUserException e) {
394            throw e;
395        } catch (Throwable e) {
396            testNotifier.addFailure(e);
397        } finally {
398            testNotifier.fireTestSuiteFinished();
399        }
400    }

402    //
403    // Implementation of Filterable and Sortable
404    //
405
406    public void filter(Filter filter) throws NoTestsRemainException {
407        childrenLock.lock();
408        try {
409            List<T> children = new ArrayList<T>(getFilteredChildren());
410            for (Iterator<T> iter = children.iterator(); iter.hasNext(); ) {
411                T each = iter.next();
412                if (shouldRun(filter, each)) {
413                    try {
414                        filter.apply(each);
415                    } catch (NoTestsRemainException e) {
416                        iter.remove();
417                    }
418                } else {
419                    iter.remove();
420                }
421            }
422            filteredChildren = Collections.unmodifiableCollection(children);
423            if (filteredChildren.isEmpty()) {
424                throw new NoTestsRemainException();
425            }
426        } finally {
427            childrenLock.unlock();
428        }
429    }
430
431    public void sort(Sorter sorter) {
432        childrenLock.lock();
433        try {
434            for (T each : getFilteredChildren()) {
435                sorter.apply(each);
436            }
437            List<T> sortedChildren = new ArrayList<T>(getFilteredChildren());
438            Collections.sort(sortedChildren, comparator(sorter));
439            filteredChildren = Collections.unmodifiableCollection(sortedChildren);
440        } finally {
441            childrenLock.unlock();
442        }
443    }

449    private void validate() throws InitializationError {
450        List<Throwable> errors = new ArrayList<Throwable>();
451        collectInitializationErrors(errors);
452        if (!errors.isEmpty()) {
453            throw new InvalidTestClassError(testClass.getJavaClass(), errors);
454        }
455    }
456
457    private Collection<T> getFilteredChildren() {
458        if (filteredChildren == null) {
459            childrenLock.lock();
460            try {
461                if (filteredChildren == null) {
462                    filteredChildren = Collections.unmodifiableCollection(getChildren());
463                }
464            } finally {
465                childrenLock.unlock();
466            }
467        }
468        return filteredChildren;
469    }
470
471    private boolean shouldRun(Filter filter, T each) {
472        return filter.shouldRun(describeChild(each));
473    }
474
475    private Comparator<? super T> comparator(final Sorter sorter) {
476        return new Comparator<T>() {
477            public int compare(T o1, T o2) {
478                return sorter.compare(describeChild(o1), describeChild(o2));
479            }
480        };
481    }

037/**
038 * Implements the JUnit 4 standard test case class model, as defined by the
039 * annotations in the org.junit package. Many users will never notice this
040 * class: it is now the default test class runner, but it should have exactly
041 * the same behavior as the old test class runner ({@code JUnit4ClassRunner}).
042 * <p>
043 * BlockJUnit4ClassRunner has advantages for writers of custom JUnit runners
044 * that are slight changes to the default behavior, however:
045 *
046 * <ul>
047 * <li>It has a much simpler implementation based on {@link Statement}s,
048 * allowing new operations to be inserted into the appropriate point in the
049 * execution flow.
050 *
051 * <li>It is published, and extension and reuse are encouraged, whereas {@code
052 * JUnit4ClassRunner} was in an internal package, and is now deprecated.
053 * </ul>
054 * <p>
055 * In turn, in 2009 we introduced {@link Rule}s.  In many cases where extending
056 * BlockJUnit4ClassRunner was necessary to add new behavior, {@link Rule}s can
057 * be used, which makes the extension more reusable and composable.
058 *
059 * @since 4.5
060 */
061public class BlockJUnit4ClassRunner extends ParentRunner<FrameworkMethod> {
062    private static TestClassValidator PUBLIC_CLASS_VALIDATOR = new PublicClassValidator();
063
064    private final ConcurrentMap<FrameworkMethod, Description> methodDescriptions = new ConcurrentHashMap<FrameworkMethod, Description>();
065

067     * Creates a BlockJUnit4ClassRunner to run {@code testClass}
068     *
069     * @throws InitializationError if the test class is malformed.
070     */

072        super(testClass);

089    @Override
090    protected void runChild(final FrameworkMethod method, RunNotifier notifier) {
091        Description description = describeChild(method);
092        if (isIgnored(method)) {
093            notifier.fireTestIgnored(description);
094        } else {
095            Statement statement = new Statement() {
096                @Override
097                public void evaluate() throws Throwable {
098                    methodBlock(method).evaluate();
099                }
100            };
101            runLeaf(statement, description, notifier);
102        }
103    }
104
105    /**
106     * Evaluates whether {@link FrameworkMethod}s are ignored based on the
107     * {@link Ignore} annotation.
108     */
109    @Override
110    protected boolean isIgnored(FrameworkMethod child) {
111        return child.getAnnotation(Ignore.class) != null;
112    }
113
114    @Override
115    protected Description describeChild(FrameworkMethod method) {
116        Description description = methodDescriptions.get(method);
117
118        if (description == null) {
119            description = Description.createTestDescription(getTestClass().getJavaClass(),
120                    testName(method), method.getAnnotations());
121            methodDescriptions.putIfAbsent(method, description);
122        }
123
124        return description;
125    }
126
127    @Override
128    protected List<FrameworkMethod> getChildren() {
129        return computeTestMethods();
130    }

136    /**
137     * Returns the methods that run tests. Default implementation returns all
138     * methods annotated with {@code @Test} on this class and superclasses that
139     * are not overridden.
140     */
141    protected List<FrameworkMethod> computeTestMethods() {
142        return getTestClass().getAnnotatedMethods(Test.class);
143    }

145    @Override
146    protected void collectInitializationErrors(List<Throwable> errors) {
147        super.collectInitializationErrors(errors);
148
149        validatePublicConstructor(errors);
150        validateNoNonStaticInnerClass(errors);
151        validateConstructor(errors);
152        validateInstanceMethods(errors);
153        validateFields(errors);
154        validateMethods(errors);
155    }
156
157    private void validatePublicConstructor(List<Throwable> errors) {
158        if (getTestClass().getJavaClass() != null) {
159            errors.addAll(PUBLIC_CLASS_VALIDATOR.validateTestClass(getTestClass()));
160        }
161    }
162
163    protected void validateNoNonStaticInnerClass(List<Throwable> errors) {
164        if (getTestClass().isANonStaticInnerClass()) {
165            String gripe = "The inner class " + getTestClass().getName()
166                    + " is not static.";
167            errors.add(new Exception(gripe));
168        }
169    }
170
171    /**
172     * Adds to {@code errors} if the test class has more than one constructor,
173     * or if the constructor takes parameters. Override if a subclass requires
174     * different validation rules.
175     */
176    protected void validateConstructor(List<Throwable> errors) {
177        validateOnlyOneConstructor(errors);
178        validateZeroArgConstructor(errors);
179    }
180
181    /**
182     * Adds to {@code errors} if the test class has more than one constructor
183     * (do not override)
184     */
185    protected void validateOnlyOneConstructor(List<Throwable> errors) {
186        if (!hasOneConstructor()) {
187            String gripe = "Test class should have exactly one public constructor";
188            errors.add(new Exception(gripe));
189        }
190    }
191
192    /**
193     * Adds to {@code errors} if the test class's single constructor takes
194     * parameters (do not override)
195     */
196    protected void validateZeroArgConstructor(List<Throwable> errors) {
197        if (!getTestClass().isANonStaticInnerClass()
198                && hasOneConstructor()
199                && (getTestClass().getOnlyConstructor().getParameterTypes().length != 0)) {
200            String gripe = "Test class should have exactly one public zero-argument constructor";
201            errors.add(new Exception(gripe));
202        }
203    }
204
205    private boolean hasOneConstructor() {
206        return getTestClass().getJavaClass().getConstructors().length == 1;
207    }
208
209    /**
210     * Adds to {@code errors} for each method annotated with {@code @Test},
211     * {@code @Before}, or {@code @After} that is not a public, void instance
212     * method with no arguments.
213     * @deprecated
214     */
215    @Deprecated
216    protected void validateInstanceMethods(List<Throwable> errors) {
217        validatePublicVoidNoArgMethods(After.class, false, errors);
218        validatePublicVoidNoArgMethods(Before.class, false, errors);
219        validateTestMethods(errors);
220
221        if (computeTestMethods().isEmpty()) {
222            errors.add(new Exception("No runnable methods"));
223        }
224    }
225
226    protected void validateFields(List<Throwable> errors) {
227        RULE_VALIDATOR.validate(getTestClass(), errors);
228    }
229
230    private void validateMethods(List<Throwable> errors) {
231        RULE_METHOD_VALIDATOR.validate(getTestClass(), errors);
232    }
233
234    /**
235     * Adds to {@code errors} for each method annotated with {@code @Test}that
236     * is not a public, void instance method with no arguments.
237     */
238    protected void validateTestMethods(List<Throwable> errors) {
239        validatePublicVoidNoArgMethods(Test.class, false, errors);
240    }

242    /**
243     * Returns a new fixture for running a test. Default implementation executes
244     * the test class's no-argument constructor (validation should have ensured
245     * one exists).
246     */
247    protected Object createTest() throws Exception {
248        return getTestClass().getOnlyConstructor().newInstance();
249    }

251    /**
252     * Returns a new fixture to run a particular test {@code method} against.
253     * Default implementation executes the no-argument {@link #createTest()} method.
254     *
255     * @since 4.13
256     */
257    protected Object createTest(FrameworkMethod method) throws Exception {
258        return createTest();
259    }

261    /**
262     * Returns the name that describes {@code method} for {@link Description}s.
263     * Default implementation is the method's name
264     */
265    protected String testName(FrameworkMethod method) {
266        return method.getName();

269    /**
270     * Returns a Statement that, when executed, either returns normally if
271     * {@code method} passes, or throws an exception if {@code method} fails.
272     *
273     * Here is an outline of the default implementation:
274     *
275     * <ul>
276     * <li>Invoke {@code method} on the result of {@link #createTest(org.junit.runners.model.FrameworkMethod)}, and
277     * throw any exceptions thrown by either operation.
278     * <li>HOWEVER, if {@code method}'s {@code @Test} annotation has the {@link Test#expected()}
279     * attribute, return normally only if the previous step threw an
280     * exception of the correct type, and throw an exception otherwise.
281     * <li>HOWEVER, if {@code method}'s {@code @Test} annotation has the {@code
282     * timeout} attribute, throw an exception if the previous step takes more
283     * than the specified number of milliseconds.
284     * <li>ALWAYS run all non-overridden {@code @Before} methods on this class
285     * and superclasses before any of the previous steps; if any throws an
286     * Exception, stop execution and pass the exception on.
287     * <li>ALWAYS run all non-overridden {@code @After} methods on this class
288     * and superclasses after any of the previous steps; all After methods are
289     * always executed: exceptions thrown by previous steps are combined, if
290     * necessary, with exceptions from After methods into a
291     * {@link MultipleFailureException}.
292     * <li>ALWAYS allow {@code @Rule} fields to modify the execution of the
293     * above steps. A {@code Rule} may prevent all execution of the above steps,
294     * or add additional behavior before and after, or modify thrown exceptions.
295     * For more information, see {@link TestRule}
296     * </ul>
297     *
298     * This can be overridden in subclasses, either by overriding this method,
299     * or the implementations creating each sub-statement.
300     */
301    protected Statement methodBlock(final FrameworkMethod method) {
302        Object test;
303        try {
304            test = new ReflectiveCallable() {
305                @Override
306                protected Object runReflectiveCall() throws Throwable {
307                    return createTest(method);
308                }
309            }.run();
310        } catch (Throwable e) {
311            return new Fail(e);
312        }
313
314        Statement statement = methodInvoker(method, test);
315        statement = possiblyExpectingExceptions(method, test, statement);
316        statement = withPotentialTimeout(method, test, statement);
317        statement = withBefores(method, test, statement);
318        statement = withAfters(method, test, statement);
319        statement = withRules(method, test, statement);
320        return statement;
321    }

327    /**
328     * Returns a {@link Statement} that invokes {@code method} on {@code test}
329     */
330    protected Statement methodInvoker(FrameworkMethod method, Object test) {
331        return new InvokeMethod(method, test);
332    }
333
334    /**
335     * Returns a {@link Statement}: if {@code method}'s {@code @Test} annotation
336     * has the {@link Test#expected()} attribute, return normally only if {@code next}
337     * throws an exception of the correct type, and throw an exception
338     * otherwise.
339     */
340    protected Statement possiblyExpectingExceptions(FrameworkMethod method,
341            Object test, Statement next) {
342        Test annotation = method.getAnnotation(Test.class);
343        Class<? extends Throwable> expectedExceptionClass = getExpectedException(annotation);
344        return expectedExceptionClass != null ? new ExpectException(next, expectedExceptionClass) : next;
345    }
346
347    /**
348     * Returns a {@link Statement}: if {@code method}'s {@code @Test} annotation
349     * has the {@code timeout} attribute, throw an exception if {@code next}
350     * takes more than the specified number of milliseconds.
351     * @deprecated
352     */
353    @Deprecated
354    protected Statement withPotentialTimeout(FrameworkMethod method,
355            Object test, Statement next) {
356        long timeout = getTimeout(method.getAnnotation(Test.class));
357        if (timeout <= 0) {
358            return next;
359        }
360        return FailOnTimeout.builder()
361               .withTimeout(timeout, TimeUnit.MILLISECONDS)
362               .build(next);
363    }
364
365    /**
366     * Returns a {@link Statement}: run all non-overridden {@code @Before}
367     * methods on this class and superclasses before running {@code next}; if
368     * any throws an Exception, stop execution and pass the exception on.
369     */
370    protected Statement withBefores(FrameworkMethod method, Object target,
371            Statement statement) {
372        List<FrameworkMethod> befores = getTestClass().getAnnotatedMethods(
373                Before.class);
374        return befores.isEmpty() ? statement : new RunBefores(statement,
375                befores, target);
376    }
377
378    /**
379     * Returns a {@link Statement}: run all non-overridden {@code @After}
380     * methods on this class and superclasses before running {@code next}; all
381     * After methods are always executed: exceptions thrown by previous steps
382     * are combined, if necessary, with exceptions from After methods into a
383     * {@link MultipleFailureException}.
384     */
385    protected Statement withAfters(FrameworkMethod method, Object target,
386            Statement statement) {
387        List<FrameworkMethod> afters = getTestClass().getAnnotatedMethods(
388                After.class);
389        return afters.isEmpty() ? statement : new RunAfters(statement, afters,
390                target);
391    }
392
393    private Statement withRules(FrameworkMethod method, Object target,
394            Statement statement) {
395        List<TestRule> testRules = getTestRules(target);
396        Statement result = statement;
397        result = withMethodRules(method, testRules, target, result);
398        result = withTestRules(method, testRules, result);
399
400        return result;
401    }
402
403    private Statement withMethodRules(FrameworkMethod method, List<TestRule> testRules,
404            Object target, Statement result) {
405        Statement withMethodRules = result;
406        for (org.junit.rules.MethodRule each : getMethodRules(target)) {
407            if (!(each instanceof TestRule && testRules.contains(each))) {
408                withMethodRules = each.apply(withMethodRules, method, target);
409            }
410        }
411        return withMethodRules;
412    }
413
414    private List<org.junit.rules.MethodRule> getMethodRules(Object target) {
415        return rules(target);
416    }
417
418    /**
419     * @param target the test case instance
420     * @return a list of MethodRules that should be applied when executing this
421     *         test
422     */
423    protected List<MethodRule> rules(Object target) {
424        List<MethodRule> rules = getTestClass().getAnnotatedMethodValues(target, 
425                Rule.class, MethodRule.class);
426
427        rules.addAll(getTestClass().getAnnotatedFieldValues(target,
428                Rule.class, MethodRule.class));
429
430        return rules;
431    }
432
433    /**
434     * Returns a {@link Statement}: apply all non-static fields
435     * annotated with {@link Rule}.
436     *
437     * @param statement The base statement
438     * @return a RunRules statement if any class-level {@link Rule}s are
439     *         found, or the base statement
440     */
441    private Statement withTestRules(FrameworkMethod method, List<TestRule> testRules,
442            Statement statement) {
443        return testRules.isEmpty() ? statement :
444                new RunRules(statement, testRules, describeChild(method));
445    }
446
447    /**
448     * @param target the test case instance
449     * @return a list of TestRules that should be applied when executing this
450     *         test
451     */
452    protected List<TestRule> getTestRules(Object target) {
453        List<TestRule> result = getTestClass().getAnnotatedMethodValues(target,
454                Rule.class, TestRule.class);
455
456        result.addAll(getTestClass().getAnnotatedFieldValues(target,
457                Rule.class, TestRule.class));
458
459        return result;
460    }
461
462    private Class<? extends Throwable> getExpectedException(Test annotation) {
463        if (annotation == null || annotation.expected() == None.class) {
464            return null;
465        } else {
466            return annotation.expected();
467        }
468    }
469
470    private long getTimeout(Test annotation) {
471        if (annotation == null) {
472            return 0;
473        }
474        return annotation.timeout();
475    }

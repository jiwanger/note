028/**
029 * The custom runner <code>Parameterized</code> implements parameterized tests.
030 * When running a parameterized test class, instances are created for the
031 * cross-product of the test methods and the test data elements.
032 * <p>
033 * For example, to test the <code>+</code> operator, write:
034 * <pre>
035 * &#064;RunWith(Parameterized.class)
036 * public class AdditionTest {
037 *     &#064;Parameters(name = &quot;{index}: {0} + {1} = {2}&quot;)
038 *     public static Iterable&lt;Object[]&gt; data() {
039 *         return Arrays.asList(new Object[][] { { 0, 0, 0 }, { 1, 1, 2 },
040 *                 { 3, 2, 5 }, { 4, 3, 7 } });
041 *     }
042 *
043 *     private int firstSummand;
044 *
045 *     private int secondSummand;
046 *
047 *     private int sum;
048 *
049 *     public AdditionTest(int firstSummand, int secondSummand, int sum) {
050 *         this.firstSummand = firstSummand;
051 *         this.secondSummand = secondSummand;
052 *         this.sum = sum;
053 *     }
054 *
055 *     &#064;Test
056 *     public void test() {
057 *         assertEquals(sum, firstSummand + secondSummand);
058 *     }
059 * }
060 * </pre>
061 * <p>
062 * Each instance of <code>AdditionTest</code> will be constructed using the
063 * three-argument constructor and the data values in the
064 * <code>&#064;Parameters</code> method.
065 * <p>
066 * In order that you can easily identify the individual tests, you may provide a
067 * name for the <code>&#064;Parameters</code> annotation. This name is allowed
068 * to contain placeholders, which are replaced at runtime. The placeholders are
069 * <dl>
070 * <dt>{index}</dt>
071 * <dd>the current parameter index</dd>
072 * <dt>{0}</dt>
073 * <dd>the first parameter value</dd>
074 * <dt>{1}</dt>
075 * <dd>the second parameter value</dd>
076 * <dt>...</dt>
077 * <dd>...</dd>
078 * </dl>
079 * <p>
080 * In the example given above, the <code>Parameterized</code> runner creates
081 * names like <code>[2: 3 + 2 = 5]</code>. If you don't use the name parameter,
082 * then the current parameter index is used as name.
083 * <p>
084 * You can also write:
085 * <pre>
086 * &#064;RunWith(Parameterized.class)
087 * public class AdditionTest {
088 *     &#064;Parameters(name = &quot;{index}: {0} + {1} = {2}&quot;)
089 *     public static Iterable&lt;Object[]&gt; data() {
090 *         return Arrays.asList(new Object[][] { { 0, 0, 0 }, { 1, 1, 2 },
091 *                 { 3, 2, 5 }, { 4, 3, 7 } });
092 *     }
093 *
094 *     &#064;Parameter(0)
095 *     public int firstSummand;
096 *
097 *     &#064;Parameter(1)
098 *     public int secondSummand;
099 *
100 *     &#064;Parameter(2)
101 *     public int sum;
102 *
103 *     &#064;Test
104 *     public void test() {
105 *         assertEquals(sum, firstSummand + secondSummand);
106 *     }
107 * }
108 * </pre>
109 * <p>
110 * Each instance of <code>AdditionTest</code> will be constructed with the default constructor
111 * and fields annotated by <code>&#064;Parameter</code>  will be initialized
112 * with the data values in the <code>&#064;Parameters</code> method.
113 *
114 * <p>
115 * The parameters can be provided as an array, too:
116 * 
117 * <pre>
118 * &#064;Parameters
119 * public static Object[][] data() {
120 *      return new Object[][] { { 0, 0, 0 }, { 1, 1, 2 }, { 3, 2, 5 }, { 4, 3, 7 } } };
121 * }
122 * </pre>
123 * 
124 * <h3>Tests with single parameter</h3>
125 * <p>
126 * If your test needs a single parameter only, you don't have to wrap it with an
127 * array. Instead you can provide an <code>Iterable</code> or an array of
128 * objects.
129 * <pre>
130 * &#064;Parameters
131 * public static Iterable&lt;? extends Object&gt; data() {
132 *      return Arrays.asList(&quot;first test&quot;, &quot;second test&quot;);
133 * }
134 * </pre>
135 * <p>
136 * or
137 * <pre>
138 * &#064;Parameters
139 * public static Object[] data() {
140 *      return new Object[] { &quot;first test&quot;, &quot;second test&quot; };
141 * }
142 * </pre>
143 *
144 * <h3>Executing code before/after executing tests for specific parameters</h3>
145 * <p>
146 * If your test needs to perform some preparation or cleanup based on the
147 * parameters, this can be done by adding public static methods annotated with
148 * {@code @BeforeParam}/{@code @AfterParam}. Such methods should either have no
149 * parameters or the same parameters as the test.
150 * <pre>
151 * &#064;BeforeParam
152 * public static void beforeTestsForParameter(String onlyParameter) {
153 *     System.out.println("Testing " + onlyParameter);
154 * }
155 * </pre>
156 *
157 * <h3>Create different runners</h3>
158 * <p>
159 * By default the {@code Parameterized} runner creates a slightly modified
160 * {@link BlockJUnit4ClassRunner} for each set of parameters. You can build an
161 * own {@code Parameterized} runner that creates another runner for each set of
162 * parameters. Therefore you have to build a {@link ParametersRunnerFactory}
163 * that creates a runner for each {@link TestWithParameters}. (
164 * {@code TestWithParameters} are bundling the parameters and the test name.)
165 * The factory must have a public zero-arg constructor.
166 *
167 * <pre>
168 * public class YourRunnerFactory implements ParametersRunnerFactory {
169 *     public Runner createRunnerForTestWithParameters(TestWithParameters test)
170 *             throws InitializationError {
171 *         return YourRunner(test);
172 *     }
173 * }
174 * </pre>
175 * <p>
176 * Use the {@link UseParametersRunnerFactory} to tell the {@code Parameterized}
177 * runner that it should use your factory.
178 *
179 * <pre>
180 * &#064;RunWith(Parameterized.class)
181 * &#064;UseParametersRunnerFactory(YourRunnerFactory.class)
182 * public class YourTest {
183 *     ...
184 * }
185 * </pre>
186 *
187 * @since 4.0
188 */
189public class Parameterized extends Suite {
190    /**
191     * Annotation for a method which provides parameters to be injected into the
192     * test class constructor by <code>Parameterized</code>. The method has to
193     * be public and static.
194     */
195    @Retention(RetentionPolicy.RUNTIME)
196    @Target(ElementType.METHOD)
197    public @interface Parameters {
198        /**
199         * Optional pattern to derive the test's name from the parameters. Use
200         * numbers in braces to refer to the parameters or the additional data
201         * as follows:
202         * <pre>
203         * {index} - the current parameter index
204         * {0} - the first parameter value
205         * {1} - the second parameter value
206         * etc...
207         * </pre>
208         * <p>
209         * Default value is "{index}" for compatibility with previous JUnit
210         * versions.
211         *
212         * @return {@link MessageFormat} pattern string, except the index
213         *         placeholder.
214         * @see MessageFormat
215         */
216        String name() default "{index}";
217    }
218
219    /**
220     * Annotation for fields of the test class which will be initialized by the
221     * method annotated by <code>Parameters</code>.
222     * By using directly this annotation, the test class constructor isn't needed.
223     * Index range must start at 0.
224     * Default value is 0.
225     */
226    @Retention(RetentionPolicy.RUNTIME)
227    @Target(ElementType.FIELD)
228    public @interface Parameter {
229        /**
230         * Method that returns the index of the parameter in the array
231         * returned by the method annotated by <code>Parameters</code>.
232         * Index range must start at 0.
233         * Default value is 0.
234         *
235         * @return the index of the parameter.
236         */
237        int value() default 0;
238    }
239
240    /**
241     * Add this annotation to your test class if you want to generate a special
242     * runner. You have to specify a {@link ParametersRunnerFactory} class that
243     * creates such runners. The factory must have a public zero-arg
244     * constructor.
245     */
246    @Retention(RetentionPolicy.RUNTIME)
247    @Inherited
248    @Target(ElementType.TYPE)
249    public @interface UseParametersRunnerFactory {
250        /**
251         * @return a {@link ParametersRunnerFactory} class (must have a default
252         *         constructor)
253         */
254        Class<? extends ParametersRunnerFactory> value() default BlockJUnit4ClassRunnerWithParametersFactory.class;
255    }
256
257    /**
258     * Annotation for {@code public static void} methods which should be executed before
259     * evaluating tests with particular parameters.
260     *
261     * @see org.junit.BeforeClass
262     * @see org.junit.Before
263     * @since 4.13
264     */
265    @Retention(RetentionPolicy.RUNTIME)
266    @Target(ElementType.METHOD)
267    public @interface BeforeParam {
268    }
269
270    /**
271     * Annotation for {@code public static void} methods which should be executed after
272     * evaluating tests with particular parameters.
273     *
274     * @see org.junit.AfterClass
275     * @see org.junit.After
276     * @since 4.13
277     */
278    @Retention(RetentionPolicy.RUNTIME)
279    @Target(ElementType.METHOD)
280    public @interface AfterParam {
281    }
282
283    /**
284     * Only called reflectively. Do not use programmatically.
285     */
286    public Parameterized(Class<?> klass) throws Throwable {
287        this(klass, new RunnersFactory(klass));
288    }
289
290    private Parameterized(Class<?> klass, RunnersFactory runnersFactory) throws Exception {
291        super(klass, runnersFactory.createRunners());
292        validateBeforeParamAndAfterParamMethods(runnersFactory.parameterCount);
293    }
294
295    private void validateBeforeParamAndAfterParamMethods(Integer parameterCount)
296            throws InvalidTestClassError {
297        List<Throwable> errors = new ArrayList<Throwable>();
298        validatePublicStaticVoidMethods(Parameterized.BeforeParam.class, parameterCount, errors);
299        validatePublicStaticVoidMethods(Parameterized.AfterParam.class, parameterCount, errors);
300        if (!errors.isEmpty()) {
301            throw new InvalidTestClassError(getTestClass().getJavaClass(), errors);
302        }
303    }
304
305    private void validatePublicStaticVoidMethods(
306            Class<? extends Annotation> annotation, Integer parameterCount,
307            List<Throwable> errors) {
308        List<FrameworkMethod> methods = getTestClass().getAnnotatedMethods(annotation);
309        for (FrameworkMethod fm : methods) {
310            fm.validatePublicVoid(true, errors);
311            if (parameterCount != null) {
312                int methodParameterCount = fm.getMethod().getParameterTypes().length;
313                if (methodParameterCount != 0 && methodParameterCount != parameterCount) {
314                    errors.add(new Exception("Method " + fm.getName()
315                            + "() should have 0 or " + parameterCount + " parameter(s)"));
316                }
317            }
318        }
319    }
320
321    private static class AssumptionViolationRunner extends Runner {
322        private final Description description;
323        private final AssumptionViolatedException exception;
324
325        AssumptionViolationRunner(TestClass testClass, String methodName,
326                AssumptionViolatedException exception) {
327            this.description = Description
328                    .createTestDescription(testClass.getJavaClass(),
329                            methodName + "() assumption violation");
330            this.exception = exception;
331        }
332
333        @Override
334        public Description getDescription() {
335            return description;
336        }
337
338        @Override
339        public void run(RunNotifier notifier) {
340            notifier.fireTestAssumptionFailed(new Failure(description, exception));
341        }
342    }
343
344    private static class RunnersFactory {
345        private static final ParametersRunnerFactory DEFAULT_FACTORY = new BlockJUnit4ClassRunnerWithParametersFactory();
346
347        private final TestClass testClass;
348        private final FrameworkMethod parametersMethod;
349        private final List<Object> allParameters;
350        private final int parameterCount;
351        private final Runner runnerOverride;
352
353        private RunnersFactory(Class<?> klass) throws Throwable {
354            testClass = new TestClass(klass);
355            parametersMethod = getParametersMethod(testClass);
356            List<Object> allParametersResult;
357            AssumptionViolationRunner assumptionViolationRunner = null;
358            try {
359                allParametersResult = allParameters(testClass, parametersMethod);
360            } catch (AssumptionViolatedException e) {
361                allParametersResult = Collections.emptyList();
362                assumptionViolationRunner = new AssumptionViolationRunner(testClass,
363                        parametersMethod.getName(), e);
364            }
365            allParameters = allParametersResult;
366            runnerOverride = assumptionViolationRunner;
367            parameterCount =
368                    allParameters.isEmpty() ? 0 : normalizeParameters(allParameters.get(0)).length;
369        }
370
371        private List<Runner> createRunners() throws Exception {
372            if (runnerOverride != null) {
373                return Collections.singletonList(runnerOverride);
374            }
375            Parameters parameters = parametersMethod.getAnnotation(Parameters.class);
376            return Collections.unmodifiableList(createRunnersForParameters(
377                    allParameters, parameters.name(),
378                    getParametersRunnerFactory()));
379        }
380
381        private ParametersRunnerFactory getParametersRunnerFactory()
382                throws InstantiationException, IllegalAccessException {
383            UseParametersRunnerFactory annotation = testClass
384                    .getAnnotation(UseParametersRunnerFactory.class);
385            if (annotation == null) {
386                return DEFAULT_FACTORY;
387            } else {
388                Class<? extends ParametersRunnerFactory> factoryClass = annotation
389                        .value();
390                return factoryClass.newInstance();
391            }
392        }
393
394        private TestWithParameters createTestWithNotNormalizedParameters(
395                String pattern, int index, Object parametersOrSingleParameter) {
396            Object[] parameters = normalizeParameters(parametersOrSingleParameter);
397            return createTestWithParameters(testClass, pattern, index, parameters);
398        }
399
400        private static Object[] normalizeParameters(Object parametersOrSingleParameter) {
401            return (parametersOrSingleParameter instanceof Object[]) ? (Object[]) parametersOrSingleParameter
402                    : new Object[] { parametersOrSingleParameter };
403        }
404
405        @SuppressWarnings("unchecked")
406        private static List<Object> allParameters(
407                TestClass testClass, FrameworkMethod parametersMethod) throws Throwable {
408            Object parameters = parametersMethod.invokeExplosively(null);
409            if (parameters instanceof List) {
410                return (List<Object>) parameters;
411            } else if (parameters instanceof Collection) {
412                return new ArrayList<Object>((Collection<Object>) parameters);
413            } else if (parameters instanceof Iterable) {
414                List<Object> result = new ArrayList<Object>();
415                for (Object entry : ((Iterable<Object>) parameters)) {
416                    result.add(entry);
417                }
418                return result;
419            } else if (parameters instanceof Object[]) {
420                return Arrays.asList((Object[]) parameters);
421            } else {
422                throw parametersMethodReturnedWrongType(testClass, parametersMethod);
423            }
424        }
425
426        private static FrameworkMethod getParametersMethod(TestClass testClass) throws Exception {
427            List<FrameworkMethod> methods = testClass
428                    .getAnnotatedMethods(Parameters.class);
429            for (FrameworkMethod each : methods) {
430                if (each.isStatic() && each.isPublic()) {
431                    return each;
432                }
433            }
434
435            throw new Exception("No public static parameters method on class "
436                    + testClass.getName());
437        }
438
439        private List<Runner> createRunnersForParameters(
440                Iterable<Object> allParameters, String namePattern,
441                ParametersRunnerFactory runnerFactory) throws Exception {
442            try {
443                List<TestWithParameters> tests = createTestsForParameters(
444                        allParameters, namePattern);
445                List<Runner> runners = new ArrayList<Runner>();
446                for (TestWithParameters test : tests) {
447                    runners.add(runnerFactory
448                            .createRunnerForTestWithParameters(test));
449                }
450                return runners;
451            } catch (ClassCastException e) {
452                throw parametersMethodReturnedWrongType(testClass, parametersMethod);
453            }
454        }
455
456        private List<TestWithParameters> createTestsForParameters(
457                Iterable<Object> allParameters, String namePattern)
458                throws Exception {
459            int i = 0;
460            List<TestWithParameters> children = new ArrayList<TestWithParameters>();
461            for (Object parametersOfSingleTest : allParameters) {
462                children.add(createTestWithNotNormalizedParameters(namePattern,
463                        i++, parametersOfSingleTest));
464            }
465            return children;
466        }
467
468        private static Exception parametersMethodReturnedWrongType(
469                TestClass testClass, FrameworkMethod parametersMethod) throws Exception {
470            String className = testClass.getName();
471            String methodName = parametersMethod.getName();
472            String message = MessageFormat.format(
473                    "{0}.{1}() must return an Iterable of arrays.", className,
474                    methodName);
475            return new Exception(message);
476        }
477
478        private TestWithParameters createTestWithParameters(
479                TestClass testClass, String pattern, int index,
480                Object[] parameters) {
481            String finalPattern = pattern.replaceAll("\\{index\\}",
482                    Integer.toString(index));
483            String name = MessageFormat.format(finalPattern, parameters);
484            return new TestWithParameters("[" + name + "]", testClass,
485                    Arrays.asList(parameters));
486        }
487    }
